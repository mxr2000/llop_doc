Statements

Variable Declaration

Example:

VAR age: Int

variableDeclStatement: VAR IDENTIFIER ':' type
;

If Statement

IF (self.salary > 1000) THEN
    self.salary = 1000
ELSE
    self.salary = 500

ifStatement: IF '(' expr ')' THEN stmt (ELSE stmt)?
;

While Statement

WHILE (self.salary > 0)
    self.salary = self.salary - 1

whileStatement: WHILE '(' expr ')' stmt
;

Return Statement

RETURN p1.getAge() - p2.getAge()

returnStatement: RETURN expr
;

Output Statement

OUTPUT 1

outputStatement: OUTPUT expr
;

Assignment

a = 10

assignStatement: expr '=' expr
;

Block Statement

BEGIN
    VAR a: Int
    a = 10
    OUTPUT a
END

blockStatement: BEGIN (stmt)* END
;

New Object

NEW Person(10, 20)

NEW referenceType argumentList
;

Number

NEW Person(10, 20)

NUMBER
