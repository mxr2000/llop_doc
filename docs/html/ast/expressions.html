Expressions

Binary Expression

Example:

a + b
a - b

multiplicativeExpr: expr op=(MUL | DIV) expr
;
additiveExpr: expr op=(ADD | SUB) expr
;

Is Expression

Example:

a is Person

isExpr: expr IS referenceType
;

Static Access

Example:

Person::compare

staticAccessExpr : referenceType '::' expr
;

Function Call

Example:

eat(b)

funCallExpr: IDENTIFIER argumentList
;

Type Coercion

Example:

person.[Employee]

typeCoercionExpr: expr'.' '[' referenceType ']'
;

Access

Example:

person.[Employee]

accessExpr: expr'.' '[' referenceType ']'
;

And

Example:

a == 10 AND b == 9

andExpr: expr AND expr
;

Or

Example:

a == 10 OR b == 9

orExpr: expr OR expr
;

Not

Example:

NOT (a > b)

notExpr: NOT
;

Relational Expr

Example:

a >= 9

relationalExpr: expr op=(EQ | NE | GT | LT | LE |GE) expr
;

New Object

NEW Person(10, 20)

newExpr: NEW referenceType argumentList
;

Number

NEW Person(10, 20)

numExpr: NUMBER
;
NUMBER : [0-9]+
